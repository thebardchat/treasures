const express = require('express');
const fs = require('fs').promises;
const path = require('path');
const crypto = require('crypto');

const app = express();
app.use(express.json());

// Storage for user contexts
const CONTEXTS_DIR = path.join(__dirname, 'contexts');
const userContexts = new Map();

// Ensure contexts directory exists
fs.mkdir(CONTEXTS_DIR, { recursive: true }).catch(console.error);

// Initialize user context
function getDefaultContext() {
  return {
    personality: 'Direct, action-oriented, family-first. Speaks plainly. Solutions over theories.',
    coreValues: ['family', 'efficiency', 'helping others', 'building legacy'],
    memories: [],
    interactions: [],
    preferences: {},
    stats: {
      totalMemories: 0,
      totalInteractions: 0,
      created: new Date().toISOString()
    }
  };
}

// Calculate memory relevance score
function calculateRelevance(content) {
  const keywordScores = {
    // Family
    'family': 5, 'tiffany': 5, 'sons': 5, 'kids': 4,
    // Business
    'business': 3, 'dispatch': 4, 'automation': 4, 'revenue': 4,
    // Tech
    'angel cloud': 5, 'shanebrain': 5, 'logibot': 4, 'ai': 3,
    // Values
    'legacy': 5, 'helping': 3, 'building': 3, 'solution': 3
  };
  
  let score = 0;
  const lower = content.toLowerCase();
  
  for (const [keyword, points] of Object.entries(keywordScores)) {
    if (lower.includes(keyword)) {
      score += points;
    }
  }
  
  return Math.min(score, 10); // Cap at 10
}

// Load user context from disk
async function loadUserContext(userId) {
  try {
    const filepath = path.join(CONTEXTS_DIR, `${userId}.json`);
    const data = await fs.readFile(filepath, 'utf-8');
    const context = JSON.parse(data);
    userContexts.set(userId, context);
    return context;
  } catch (error) {
    const context = getDefaultContext();
    userContexts.set(userId, context);
    return context;
  }
}

// Save user context to disk
async function saveUserContext(userId, context) {
  try {
    const filepath = path.join(CONTEXTS_DIR, `${userId}.json`);
    await fs.writeFile(filepath, JSON.stringify(context, null, 2));
    return true;
  } catch (error) {
    console.error('Save context error:', error);
    return false;
  }
}

// Learn from new content
app.post('/learn', async (req, res) => {
  try {
    const { userId, content, source, interaction } = req.body;
    
    if (!userId) {
      return res.status(400).json({ error: 'userId required' });
    }
    
    // Get or create user context
    let context = userContexts.get(userId);
    if (!context) {
      context = await loadUserContext(userId);
    }
    
    // Add memory if content provided
    if (content) {
      const memory = {
        id: crypto.randomUUID(),
        content,
        source: source || 'unknown',
        relevance: calculateRelevance(content),
        timestamp: new Date().toISOString(),
        embeddings: null // TODO: Add vector embeddings for similarity search
      };
      
      context.memories.push(memory);
      context.stats.totalMemories++;
      
      // Keep only top 1000 memories by relevance
      if (context.memories.length > 1000) {
        context.memories.sort((a, b) => b.relevance - a.relevance);
        context.memories = context.memories.slice(0, 1000);
      }
    }
    
    // Add interaction if provided
    if (interaction) {
      const record = {
        id: crypto.randomUUID(),
        message: interaction.message,
        response: interaction.response,
        timestamp: new Date().toISOString()
      };
      
      context.interactions.push(record);
      context.stats.totalInteractions++;
      
      // Keep last 100 interactions
      if (context.interactions.length > 100) {
        context.interactions = context.interactions.slice(-100);
      }
    }
    
    // Save to disk
    await saveUserContext(userId, context);
    
    res.json({
      success: true,
      stats: {
        totalMemories: context.stats.totalMemories,
        activeMemories: context.memories.length,
        totalInteractions: context.stats.totalInteractions
      }
    });
  } catch (error) {
    console.error('Learn error:', error);
    res.status(500).json({ error: 'Learning failed' });
  }
});

// Get user context (for AI to use)
app.get('/context/:userId', async (req, res) => {
  try {
    const { userId } = req.params;
    
    // Get or load context
    let context = userContexts.get(userId);
    if (!context) {
      context = await loadUserContext(userId);
    }
    
    // Build context string for AI
    const recentMemories = context.memories
      .sort((a, b) => b.relevance - a.relevance)
      .slice(0, 10)
      .map(m => `- ${m.content} (${m.source})`)
      .join('\n');
    
    const recentInteractions = context.interactions
      .slice(-5)
      .map(i => `User: ${i.message}\nShane: ${i.response}`)
      .join('\n\n');
    
    const contextString = `
PERSONALITY: ${context.personality}

CORE VALUES: ${context.coreValues.join(', ')}

RECENT RELEVANT MEMORIES (${context.memories.length} total):
${recentMemories || 'No memories yet'}

RECENT CONVERSATION STYLE (${context.interactions.length} total):
${recentInteractions || 'No interactions yet'}
    `.trim();
    
    res.json({
      context: contextString,
      raw: {
        personality: context.personality,
        coreValues: context.coreValues,
        memoriesCount: context.memories.length,
        interactionsCount: context.interactions.length
      }
    });
  } catch (error) {
    console.error('Context retrieval error:', error);
    res.status(500).json({ error: 'Context retrieval failed' });
  }
});

// Update personality/preferences
app.post('/update', async (req, res) => {
  try {
    const { userId, personality, coreValues, preferences } = req.body;
    
    if (!userId) {
      return res.status(400).json({ error: 'userId required' });
    }
    
    let context = userContexts.get(userId);
    if (!context) {
      context = await loadUserContext(userId);
    }
    
    if (personality) context.personality = personality;
    if (coreValues) context.coreValues = coreValues;
    if (preferences) context.preferences = { ...context.preferences, ...preferences };
    
    await saveUserContext(userId, context);
    
    res.json({ success: true });
  } catch (error) {
    console.error('Update error:', error);
    res.status(500).json({ error: 'Update failed' });
  }
});

// Get user stats
app.get('/stats/:userId', async (req, res) => {
  try {
    const { userId } = req.params;
    
    let context = userContexts.get(userId);
    if (!context) {
      context = await loadUserContext(userId);
    }
    
    res.json(context.stats);
  } catch (error) {
    res.status(500).json({ error: 'Stats retrieval failed' });
  }
});

// Health check
app.get('/health', (req, res) => {
  res.json({ 
    status: 'online', 
    service: 'ShaneBrain Core',
    activeUsers: userContexts.size
  });
});

const PORT = 3003;
app.listen(PORT, () => {
  console.log(`
╔══════════════════════════════════════════════════════════╗
║           SHANEBRAIN CORE - ONLINE                        ║
╚══════════════════════════════════════════════════════════╝

   Brain Service: http://localhost:${PORT}
   Contexts Dir: ${CONTEXTS_DIR}

   Endpoints:
   - POST /learn          (add memories/interactions)
   - GET  /context/:id    (get AI context)
   - POST /update         (update personality)
   - GET  /stats/:id      (get user stats)
   - GET  /health         (health check)

   Ready to learn!
`);
});

// Add this to daily-memory-updated to send memories to brain

const axios = require('axios');

const BRAIN_SERVICE = 'http://localhost:3003';

async function sendToBrain(userId, memory, source) {
  try {
    const response = await axios.post(`${BRAIN_SERVICE}/learn`, {
      userId,
      content: memory,
      source: source || 'daily-memory'
    });
    
    console.log('✓ Memory sent to ShaneBrain:', response.data.stats);
    return response.data;
  } catch (error) {
    console.error('✗ Failed to send to brain:', error.message);
    return null;
  }
}
