FILE 5: Database Migration (PostgreSQL)
==================================================

Install PostgreSQL:
1. Download from: https://www.postgresql.org/download/windows/
2. Run installer
3. Set password (remember it!)
4. Keep default port 5432

Create database:
psql -U postgres
CREATE DATABASE angelcloud;
\q

Create tables:

-- users table
CREATE TABLE users (
  id VARCHAR(255) PRIMARY KEY,
  email VARCHAR(255) UNIQUE NOT NULL,
  password VARCHAR(255) NOT NULL,
  tier VARCHAR(50) DEFAULT 'free',
  customer_id VARCHAR(255),
  created_at TIMESTAMP DEFAULT NOW(),
  updated_at TIMESTAMP DEFAULT NOW()
);

-- usage table
CREATE TABLE usage (
  id SERIAL PRIMARY KEY,
  user_id VARCHAR(255) REFERENCES users(id),
  action VARCHAR(100),
  amount INT DEFAULT 1,
  timestamp TIMESTAMP DEFAULT NOW()
);

-- memories table
CREATE TABLE memories (
  id VARCHAR(255) PRIMARY KEY,
  user_id VARCHAR(255) REFERENCES users(id),
  content TEXT,
  source VARCHAR(100),
  relevance INT,
  created_at TIMESTAMP DEFAULT NOW()
);

-- automations table
CREATE TABLE automations (
  id SERIAL PRIMARY KEY,
  user_id VARCHAR(255) REFERENCES users(id),
  rule_name VARCHAR(255),
  trigger_type VARCHAR(100),
  result TEXT,
  executed_at TIMESTAMP DEFAULT NOW()
);

FILE 6: Logging Setup
==================================================

Install Winston:
npm install winston --save

Create logger.js in each service:

const winston = require('winston');

const logger = winston.createLogger({
  level: process.env.LOG_LEVEL || 'info',
  format: winston.format.combine(
    winston.format.timestamp(),
    winston.format.json()
  ),
  transports: [
    new winston.transports.File({ 
      filename: 'error.log', 
      level: 'error' 
    }),
    new winston.transports.File({ 
      filename: 'combined.log' 
    })
  ]
});

if (process.env.NODE_ENV !== 'production') {
  logger.add(new winston.transports.Console({
    format: winston.format.simple()
  }));
}

module.exports = logger;

Usage in server.js:
const logger = require('./logger');
logger.info('Server started');
logger.error('Something failed', { error: err });

FILE 7: Health Check Monitoring
==================================================

Create: health-monitor.js

const axios = require('axios');

const services = [
  { name: 'Auth', url: 'http://localhost:3001/health' },
  { name: 'Gateway', url: 'http://localhost:3000/health' },
  { name: 'Brain', url: 'http://localhost:3003/health' },
  { name: 'LogiBot', url: 'http://localhost:3004/health' },
  { name: 'Payment', url: 'http://localhost:3006/health' },
  { name: 'Dashboard', url: 'http://localhost:3007/health' }
];

async function checkHealth() {
  console.log('\n🏥 Health Check:', new Date().toLocaleTimeString());
  
  for (const service of services) {
    try {
      await axios.get(service.url, { timeout: 2000 });
      console.log(`  ✓ ${service.name}: Online`);
    } catch (error) {
      console.log(`  ✗ ${service.name}: OFFLINE`);
      // Send alert (email, SMS, etc.)
    }
  }
}

// Check every 5 minutes
setInterval(checkHealth, 5 * 60 * 1000);
checkHealth();

Run with PM2:
pm2 start health-monitor.js --name health-monitor

FILE 10: Performance Optimization
==================================================

1. Enable Redis caching:
   npm install redis

   Cache frequently accessed data:
   - User contexts
   - Subscription status
   - Usage limits

2. Database indexing:
   CREATE INDEX idx_user_email ON users(email);
   CREATE INDEX idx_usage_user_id ON usage(user_id);
   CREATE INDEX idx_memories_user_id ON memories(user_id);

3. API Gateway rate limiting:
   npm install express-rate-limit

   const rateLimit = require('express-rate-limit');
   
   const limiter = rateLimit({
     windowMs: 15 * 60 * 1000, // 15 minutes
     max: 100 // limit each IP to 100 requests per windowMs
   });
   
   app.use(limiter);
