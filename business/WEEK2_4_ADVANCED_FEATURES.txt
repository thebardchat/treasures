const { pipeline } = require('@xenova/transformers');

class EmbeddingService {
  constructor() {
    this.extractor = null;
  }

  async initialize() {
    this.extractor = await pipeline(
      'feature-extraction',
      'Xenova/all-MiniLM-L6-v2'
    );
    console.log('✓ Embedding model loaded');
  }

  async generateEmbedding(text) {
    if (!this.extractor) await this.initialize();
    
    const output = await this.extractor(text, {
      pooling: 'mean',
      normalize: true
    });
    
    return Array.from(output.data);
  }

  cosineSimilarity(a, b) {
    let dotProduct = 0;
    let normA = 0;
    let normB = 0;
    
    for (let i = 0; i < a.length; i++) {
      dotProduct += a[i] * b[i];
      normA += a[i] * a[i];
      normB += b[i] * b[i];
    }
    
    return dotProduct / (Math.sqrt(normA) * Math.sqrt(normB));
  }

  async findSimilarMemories(query, memories, topK = 5) {
    const queryEmbedding = await this.generateEmbedding(query);
    
    const scored = await Promise.all(
      memories.map(async (memory) => {
        if (!memory.embedding) {
          memory.embedding = await this.generateEmbedding(memory.content);
        }
        
        const similarity = this.cosineSimilarity(
          queryEmbedding,
          memory.embedding
        );
        
        return { ...memory, similarity };
      })
    );
    
    return scored
      .sort((a, b) => b.similarity - a.similarity)
      .slice(0, topK);
  }
}

module.exports = new EmbeddingService();

class ConversationManager {
  constructor() {
    this.sessions = new Map();
  }

  createSession(userId) {
    const sessionId = `${userId}-${Date.now()}`;
    this.sessions.set(sessionId, {
      userId,
      messages: [],
      context: null,
      started: new Date(),
      lastActivity: new Date()
    });
    return sessionId;
  }

  addMessage(sessionId, role, content) {
    const session = this.sessions.get(sessionId);
    if (!session) return false;
    
    session.messages.push({
      role,
      content,
      timestamp: new Date()
    });
    session.lastActivity = new Date();
    
    // Keep last 50 messages
    if (session.messages.length > 50) {
      session.messages = session.messages.slice(-50);
    }
    
    return true;
  }

  getContext(sessionId, maxMessages = 10) {
    const session = this.sessions.get(sessionId);
    if (!session) return null;
    
    const recent = session.messages.slice(-maxMessages);
    
    return {
      history: recent,
      messageCount: session.messages.length,
      duration: Date.now() - session.started.getTime()
    };
  }

  endSession(sessionId) {
    this.sessions.delete(sessionId);
  }

  cleanupStale(maxAgeMinutes = 30) {
    const now = Date.now();
    for (const [id, session] of this.sessions.entries()) {
      const age = now - session.lastActivity.getTime();
      if (age > maxAgeMinutes * 60 * 1000) {
        this.sessions.delete(id);
      }
    }
  }
}

module.exports = new ConversationManager();

class WorkspaceManager {
  constructor() {
    this.workspaces = new Map();
  }

  createWorkspace(ownerId, name) {
    const workspaceId = crypto.randomUUID();
    
    this.workspaces.set(workspaceId, {
      id: workspaceId,
      name,
      ownerId,
      members: [ownerId],
      settings: {
        sharedMemories: false,
        sharedAutomations: true,
        tier: 'free'
      },
      created: new Date()
    });
    
    return workspaceId;
  }

  addMember(workspaceId, userId, role = 'member') {
    const workspace = this.workspaces.get(workspaceId);
    if (!workspace) return false;
    
    if (!workspace.members.includes(userId)) {
      workspace.members.push(userId);
    }
    
    return true;
  }

  getWorkspace(workspaceId) {
    return this.workspaces.get(workspaceId);
  }

  getUserWorkspaces(userId) {
    return Array.from(this.workspaces.values())
      .filter(w => w.members.includes(userId));
  }
}

class AnalyticsEngine {
  async generateInsights(userId) {
    const context = await getUserContext(userId);
    
    const insights = {
      productivity: this.analyzeProductivity(context),
      patterns: this.detectPatterns(context),
      recommendations: this.generateRecommendations(context)
    };
    
    return insights;
  }

  analyzeProductivity(context) {
    const automations = context.automations || [];
    const timeSaved = automations.length * 5; // 5 min per automation
    
    return {
      automationsRun: automations.length,
      estimatedTimeSaved: timeSaved,
      efficiencyScore: Math.min(100, automations.length * 2)
    };
  }

  detectPatterns(context) {
    const memories = context.memories || [];
    
    // Find common themes
    const themes = {};
    memories.forEach(m => {
      const words = m.content.toLowerCase().split(' ');
      words.forEach(word => {
        if (word.length > 5) {
          themes[word] = (themes[word] || 0) + 1;
        }
      });
    });
    
    const topThemes = Object.entries(themes)
      .sort((a, b) => b[1] - a[1])
      .slice(0, 5)
      .map(([theme]) => theme);
    
    return { topThemes };
  }

  generateRecommendations(context) {
    const recommendations = [];
    
    if (context.automations?.length < 3) {
      recommendations.push({
        type: 'automation',
        message: 'Set up more automations to save time',
        action: '/logibot/setup'
      });
    }
    
    if (context.memories?.length < 10) {
      recommendations.push({
        type: 'memory',
        message: 'Add more memories to personalize your AI',
        action: '/brain/add-memory'
      });
    }
    
    return recommendations;
  }
}
